  SELECT 
        ic.id_usuario,
        ic.id_curso,
        c.titulo AS curso,
        SUM(v.duracion_segundos) AS duracion_total,
        SUM(LEAST(pv.segundos_vistos, v.duracion_segundos)) AS segundos_vistos,
        ROUND(
            (SUM(LEAST(pv.segundos_vistos, v.duracion_segundos)) / SUM(v.duracion_segundos)) * 100,
            2
        ) AS porcentaje_avance
    FROM inscripciones_cursos ic
    JOIN cursos c ON ic.id_curso = c.id
    JOIN modulos_curso mc ON c.id = mc.id_curso
    JOIN videos v ON mc.id = v.id_modulo
    LEFT JOIN progreso_videos pv 
      ON v.id = pv.id_video 
      AND pv.id_usuario = ic.id_usuario
    WHERE ic.id_usuario = 1
    GROUP BY ic.id_usuario, ic.id_curso;
